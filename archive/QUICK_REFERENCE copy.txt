┌──────────────────────────────────────────────────────────────────────────────┐
│ PAYMENT FLOW TIMELINE                                                        │
├──────────────────────────────────────────────────────────────────────────────┤
│                                                                               │
│  T+0s    Customer enters phone number                                        │
│  T+0s    Frontend: POST /api/payments/create                                 │
│  T+0s    Backend: Insert payment record (status="pending")                   │
│  T+0s    Frontend: Show "Check your phone for PIN"                           │
│  T+2s    Frontend: Start polling GET /api/payments/status/{txn}              │
│  T+5s    Customer: Confirms PIN on phone                                     │
│  T+7s    Frontend: Poll receives "status": "successful"                      │
│  T+7s    Backend: Update payment status="completed"                          │
│  T+7s    Backend: revenue += 100, is_occupied=1                              │
│  T+7s    Backend: Log payment event (triggers SSE)                           │
│  T+7s    Frontend: Show "✅ DOOR OPENING..."                                 │
│  T+7s    Dashboard: Refresh dashboard (SSE listener)                         │
│  T+8s    ESP32: Polls GET /api/hardware/payment-check/1                      │
│  T+8s    ESP32: Receives "command": "OPEN_DOOR"                              │
│  T+8s    ESP32: Opens door (servo 0° → 90°)                                  │
│  T+18s   ESP32: Closes door (servo 90° → 0°)                                 │
│  T+18s   ESp32: Sets is_occupied=1, transitions to S_OCCUPIED               │
│  T+25s   Customer: Presses button to exit                                    │
│  T+25s   ESP32: Opens door (0° → 90°)                                        │
│  T+35s   ESP32: Closes door (90° → 0°)                                       │
│  T+35s   ESP32: Sets is_occupied=0, transitions to S_IDLE                   │
│  T+35s   Dashboard: Toilet marked FREE                                       │
│                                                                               │
└──────────────────────────────────────────────────────────────────────────────┘

┌──────────────────────────────────────────────────────────────────────────────┐
│ KEY DATABASE UPDATES                                                         │
├──────────────────────────────────────────────────────────────────────────────┤
│                                                                               │
│  PAYMENTS TABLE:                                                              │
│  ✅ INSERT new record (toilet_id, amount, phone_number, status="pending")    │
│  ✅ UPDATE status="completed" (when payment confirmed)                       │
│  ✅ UPDATE paid_at=NOW() (when payment confirmed)                            │
│  ✅ UPDATE consumed=1 (when ESP32 opens door)                                │
│                                                                               │
│  TOILETS TABLE:                                                               │
│  ✅ UPDATE revenue = revenue + amount (when payment confirmed)                │
│  ✅ UPDATE is_occupied = 1 (when payment confirmed)                           │
│  ✅ UPDATE is_occupied = 1 (when ESP32 opens door)                            │
│  ✅ UPDATE is_occupied = 0 (when customer exits)                              │
│                                                                               │
│  SENSOR_EVENTS TABLE:                                                         │
│  ✅ INSERT event_type="payment", details="..."                               │
│  ✅ INSERT event_type="payment_trigger", details="..."                       │
│  ✅ INSERT event_type="door_open", details="entry"                           │
│  ✅ INSERT event_type="door_close", details="exit complete"                  │
│                                                                               │

┌──────────────────────────────────────────────────────────────────────────────┐
│ API ENDPOINTS USED                                                           │
├──────────────────────────────────────────────────────────────────────────────┤
│                                                                               │
│  1. Frontend: POST /api/payments/create                                       │
│     Body: {toilet_id, amount, phone_number}                                  │
│     Returns: {transaction_id, payment_id, status}                            │
│                                                                               │
│  2. Frontend: GET /api/payments/status/{transaction_id}                       │
│     Returns: {status, command:"OPEN_DOOR"|"DENY", amount}                    │
│     Polling: Every 2 seconds                                                  │
│                                                                               │
│  3. ESP32: GET /api/hardware/payment-check/{toilet_id}                        │
│     Returns: {command:"OPEN_DOOR"|"DENY", message, amount}                   │
│     Polling: Every 2 seconds                                                  │
│                                                                               │
│  4. ESP32: PUT /api/hardware/occupancy/{toilet_id}                            │
│     Body: {is_occupied: true|false}                                          │
│     Returns: {success, is_occupied}                                          │
│                                                                               │
│  5. ESP32: POST /api/hardware/log-event                                       │
│     Body: {toilet_id, event_type, details}                                   │
│     Returns: {message}                                                        │
│                                                                               │
│  6. Owner: GET /api/owner/dashboard                                           │
│     Returns: {totalRevenue, toiletCount, cleanerCount, toilets}              │
│                                                                               │
│  7. Owner: GET /api/public/toilets                                            │
│     Returns: [{id, location, is_occupied}, ...]                              │
│                                                                               │
